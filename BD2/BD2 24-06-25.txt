----1-----/---//---/--------------------/---//---/--------------------/---//---/--------------------/---//---/--------------------/---//---/--------------------/---//---/----------
SELECT 
    metal.`Nome`, 
    cotacao.`Data`, 
    cotacao.`ValorKg` 
FROM 
    cotacao, 
    metal 
WHERE 
    metal.`Simbolo`= cotacao.`FK_Metal_Simbolo` 
AND 
    metal.nome = 'Ouro' 
ORDER BY 
    cotacao.`ValorKg` DESC
LIMIT 1;

----2-----/---//---/--------------------/---//---/--------------------/---//---/--------------------/---//---/--------------------/---//---/--------------------/---//---/----------
SELECT 
    metal.`Nome`, 
    cotacao.`Data`, 
    cotacao.`ValorKg` 
FROM 
    metal 
    INNER JOIN
    cotacao ON metal.`Simbolo` = cotacao.`FK_Metal_Simbolo`
WHERE 
    metal.`Simbolo`= cotacao.`FK_Metal_Simbolo` 
AND 
    metal.nome = 'Ouro' 
ORDER BY 
    cotacao.`ValorKg` DESC
LIMIT 1;

    m.`Nome`,
    c.`Data`,
    c.`ValorKG`,
    m.`Simbolo`
FROM    
    metal as m
        INNER JOIN
    cotacao as c ON m.`Simbolo` = c.`FK_Metal_Simbolo`
WHERE
    m.`Nome` = 'Cobre' AND
    c.`ValorKg` = (SELECT MAX(cotacao.`ValorKg`) FROM cotacao WHERE `FK_Metal_Simbolo` = m.`Simbolo`)
